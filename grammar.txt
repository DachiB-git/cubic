prog -> TyDS VaDS FuDS
TyDS -> TyD TyDS | epsilon
TyD -> TYPEDEF TE ;
TE -> Ty TyDeco TYNAME | STRUCT TYNAME { VaDS }
TyDeco -> PDeco ArrDeco | epsilon
PDeco -> * PDeco | epsilon
ArrDeco -> [ NUM ] ArrDeco | epsilon
Ty -> CHAR | SHORT | INT | UCHAR | USHORT | UINT | BOOL | TYNAME
VaDS -> VaD VaDS | epsilon
VaD -> Ty TyDeco NAME ;
FuDS -> FuD RFuDS
RFuDS -> FuD RFuDS | epsilon
FuD -> FUNC Ty TyDeco FUNCNAME ( PaDS ) { VaDS body }
PaD -> Ty PaDTyDeco NAME
PaDTyDeco -> PDeco PaDArrDeco | epsilon
PaDArrDeco -> [ NUM ] ArrDeco | epsilon
PaDS -> PaD RPaDS | epsilon
RPaDS -> , PaD RPaDS | epsilon
body -> StS rSt
rSt -> RETURN GenE ;
StS -> St StS | epsilon
St -> IF ( GenE ) { StS } MatchedElse | WHILE ( GenE ) { StS } | DO { StS } WHILE ( GenE ) ; | id = GenE ; | FUNCNAME ( ArgS ) ; | ASM ( STRLIT ) ;
MatchedElse -> ELSE { StS } | epsilon
GenE -> FUNCNAME ( ArgS ) | JointE | STRLIT
Arg -> GenE
ArgS -> Arg RArgS | epsilon
RArgS -> , Arg RArgS | epsilon
id -> NAME Rid
Rid -> idSel Rid | epsilon
idSel -> . NAME | [ GenE ]
E -> T RE
RE -> + T RE | - T RE | epsilon
T -> F RT
RT -> * F RT | / F RT | epsilon
F -> id | - F | ! F | * F | & F | Ty PDeco ! F | ( RelE ) | NUM | TRUE | FALSE
RelE -> E RRelE
RRelE -> RelOp E RRelE | epsilon
RelOp -> GT | GE | LT | LE | EQ | NE
JointE -> RelE Union
Union -> Intersect UnionT | epsilon
UnionT -> OR_OP RelE Intersect UnionT | epsilon
Intersect -> AND_OP RelE Intersect | epsilon
